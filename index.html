function setSize() { const wrap = document.getElementById('wheelContainer'); // 親要素の横幅から余白を引いた、表示可能な最大幅を計算（最大値を 500 に引き上げ） const avail = Math.min(wrap.parentElement.offsetWidth - 16, 500); SIZE = Math.max(280, avail); // 画面のピクセル密度（DPI）を取得 const dpr = window.devicePixelRatio || 1; // 1. Canvasの「ブラウザ上の見た目のサイズ（CSS）」をしっかり固定 canvas.style.width = SIZE + 'px'; canvas.style.height = SIZE + 'px'; // 2. Canvasの「内部の解像度（画素数）」をDPI倍にして高画質化 canvas.width = SIZE * dpr; canvas.height = SIZE * dpr; // 3. 描画コンテキストのスケールをリセットしてから再設定 ctx.setTransform(1, 0, 0, 1, 0, 0); ctx.scale(dpr, dpr); drawWheel(); }
